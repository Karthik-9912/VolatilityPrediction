#function to load dataset
def getXY(dataset_name,asset_name,typeValue):
    dataset = pd.read_csv(dataset_name)
    dataset["Date"]= pd.to_datetime(dataset["Date"])
    dataset["Year"]= dataset['Date'].dt.year
    data = dataset
    temp = dataset.values
    Y = temp[:,5:6]
    dataset = dataset.drop(["Adj Close"], axis=1)
    dataset = dataset.drop(["Volume"], axis=1)
    dataset = dataset.values
    X = dataset[:,1:dataset.shape[1]]
    X = sc.fit_transform(X)
    Y = sc.fit_transform(Y)
    XX = []
    YY = []
    for i in range(10, 250):
        XX.append(X[i-10:i, 0:X.shape[1]])
        YY.append(Y[i, 0])    
    XX, YY = np.array(XX), np.array(YY)
    if typeValue == 0:
        plt.figure(figsize=(16,10), dpi=100)
        plt.plot(data.Date[0:10], data.Close[0:10], color='tab:red')
        plt.gca().set(title=asset_name+" Adjacent Closing Prices", xlabel='Date', ylabel="Closing Price")
        plt.show()
    return data, XX, YY

def getModel():
    if os.path.exists('model/lstm_model.json'):
        with open('model/lstm_model.json', "r") as json_file:
            loaded_model_json = json_file.read()
            lstm_model = model_from_json(loaded_model_json)
        json_file.close()
        lstm_model.load_weights("model/lstm_model_weights.h5")
        lstm_model._make_predict_function()   
    else:
        #training with LSTM algorithm and saving trained model and LSTM refrence assigned to lstm_model variable
        lstm_model = Sequential()
        lstm_model.add(LSTM(units = 50, return_sequences = True, input_shape = (X_train.shape[1], X_train.shape[2])))
        lstm_model.add(Dropout(0.2))
        lstm_model.add(LSTM(units = 50, return_sequences = True))
        lstm_model.add(Dropout(0.2))
        lstm_model.add(LSTM(units = 50, return_sequences = True))
        lstm_model.add(Dropout(0.2))
        lstm_model.add(LSTM(units = 50))
        lstm_model.add(Dropout(0.2))
        lstm_model.add(Dense(units = 1))
        lstm_model.compile(optimizer = 'adam', loss = 'mean_squared_error')
        lstm_model.fit(X_train, y_train, epochs = 1000, batch_size = 8)
        lstm_model.save_weights('model/lstm_model_weights.h5')            
        model_json = lstm_model.to_json()
        with open("model/lstm_model.json", "w") as json_file:
            json_file.write(model_json)
        json_file.close()
    return lstm_model

#this function will perform prediction on test data and return Mean square value as the difference between
#original test data and predicted values from LSTM
def forecastPortfolio(lstm_model, XX, YY, asset_name):
    X_train, X_test, y_train, y_test = train_test_split(XX, YY, test_size=0.2)#getTrainTest(XX, YY)
    #performing prediction on test data    
    predict_portfolio = lstm_model.predict(X_test)
    predict_portfolio = sc.inverse_transform(predict_portfolio)
    predict_portfolio = predict_portfolio.ravel()
    y_test = y_test.reshape(y_test.shape[0],1)
    labels = sc.inverse_transform(y_test)
    labels = labels.ravel()
    
    print("\n"+asset_name+" portfolio prediction output\n")
    for i in range(len(labels)):
        print("Original Portfolio Value: "+str(labels[i])+" LSTM Predicted Portfolio Value: "+str(predict_portfolio[i]))
    
    #calculating LSTM MSE
    lstm_rmse = mean_squared_error(labels,predict_portfolio)
    print("\nLSTM Root Mean Square Error on "+asset_name+": "+str(lstm_rmse))
    #plotting LSTM predicted and original values
    plt.plot(labels, color = 'red', label = 'Original Portfolio Value')
    plt.plot(predict_portfolio, color = 'green', label = 'LSTM Predicted Portfolio Value')
    plt.title(asset_name+' LSTM Portfolio Prices Forecasting')
    plt.xlabel('Test Data')
    plt.ylabel('Forecasting Prices')
    plt.legend()
    plt.show()
